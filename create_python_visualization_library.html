import tkinter as tk

class ClassDiagram:
    def __init__(self, master):
        self.master = master
        self.canvas = tk.Canvas(self.master, width=400, height=300)
        self.canvas.pack()

        # Draw a parallelogram for the class
        self.class_rect = self.canvas.create_rectangle(
            50, 50, 150, 100, fill="lightblue"
        )

        # Add a label for the class name
        self.class_label = self.canvas.create_text(
            100, 75, text="MyClass", font=("Arial", 14)
        )

        # Draw a circle for the method
        self.method_circle = self.canvas.create_oval(
            200, 50, 250, 100, fill="pink"
        )

        # Add a label for the method name
        self.method_label = self.canvas.create_text(
            225, 75, text="my_method", font=("Arial", 14)
        )

        # Draw a circle for the variable
        self.var_circle = self.canvas.create_oval(
            300, 50, 350, 100, fill="lightgreen"
        )

        # Add a label for the variable name and value
        self.var_label = self.canvas.create_text(
            325, 75, text="my_var = 5", font=("Arial", 14)
        )

        self.canvas.create_line(125, 75, 225, 75)

        # Use a line to connect the class to the variable

import tkinter as tk
import ast

class ClassDiagram:
    def __init__(self, master, code):
        self.master = master
        self.canvas = tk.Canvas(self.master, width=400, height=300)
        self.canvas.pack()

        self.draw_diagram(code)

    def draw_diagram(self, code):
        # Parse the code using the ast module
        root = ast.parse(code)

        # Iterate through the AST nodes and draw the diagram elements
        for node in ast.iter_child_nodes(root):
            if isinstance(node, ast.ClassDef):
                # Draw a parallelogram for the class
                self.class_rect = self.canvas.create_rectangle(
                    50, 50, 150, 100, fill="lightblue"
                )

                # Add a label for the class name
                self.class_label = self.canvas.create_text(
                    100, 75, text=node.name, font=("Arial", 14)
                )

                # Iterate through the class body and draw the methods and variables
                for body_node in node.body:
                    if isinstance(body_node, ast.FunctionDef):
                        # Draw a circle for the method
                        self.method_circle = self.canvas.create_oval(
                            200, 50, 250, 100, fill="pink"
                        )

                        # Add a label for the method name
                        self.method_label = self.canvas.create_text(
                            225, 75, text=body_node.name, font=("Arial", 14)
                        )

                        # Use a line to connect the class to the method
                        self.canvas.create_line(125, 75, 225, 75)
                    elif isinstance(body_node, ast.Assign):
                        # Draw a circle for the variable
                        self.var_circle = self.canvas.create_oval(
                            300, 50, 350, 100, fill="lightgreen"
                        )

                        # Add a label for the variable name and value
                        self.var_label = self.canvas.create_text(
                            325, 75, text=f"{body_node.targets[0].id} = {body_node.value}", font=("Arial", 14)
                        )

                        # Use a line to connect the class to the variable
                        self.canvas.create_line(175, 75, 325, 75)

    # Parse the code using the ast module
